import csv
from fpdf import FPDF

# Configuration for invoice output
def generate_invoice(data):
    pdf = FPDF()
    pdf.add_page()
    pdf.set_font("Arial", size=12)
    
    # Iterate through row data
    for row in data:
        pdf.cell(200, 10, txt=row['client_name'], ln=1, align='C')
        
    pdf.output("invoice.pdf")
